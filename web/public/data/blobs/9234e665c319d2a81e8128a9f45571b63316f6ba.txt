    <meta name="viewport" content="width=device-width, initial-scale=1.0, viewport-fit=cover" />
    <meta name="theme-color" content="#ffffff" />
    <script>
      (() => {
        try {
          const stored =
            localStorage.getItem("nanocodex-theme") ?? localStorage.getItem("xedoc-theme");
          const theme = stored === "dark" ? "dark" : "light";
          document.documentElement.dataset.theme = theme;
          document
            .querySelector('meta[name="theme-color"]')
            ?.setAttribute("content", theme === "dark" ? "#161616" : "#ffffff");
        } catch {
          document.documentElement.dataset.theme = "light";
        }
      })();
    </script>
    <meta
      name="description"
      content="A from-scratch Codex rewrite for the latest models, built with the same tools and an eval-driven development loop."
